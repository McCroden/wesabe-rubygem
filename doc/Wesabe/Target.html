def self.from_xml(xml)
  new do |target|
    target.tag = xml.at("tag").at("name").inner_text
    target.monthly_limit = xml.at("monthly-limit").inner_text.to_f
    target.amount_remaining = xml.at("amount-remaining").inner_text.to_f
  end
end
