def self.from_xml(xml)
  new do |currency|
    currency.decimal_places = xml[:decimal_places].to_s.to_i
    currency.symbol         = xml[:symbol].to_s
    currency.separator      = xml[:separator].to_s
    currency.delimiter      = xml[:delimiter].to_s
  end
end
