def accounts
  unless @accounts
    @accounts = []
    accounts = Request.execute(
                :url => '/accounts.xml', 
                :username => username, 
                :password => password)
    REXML::Document.new(accounts).root.each_element("//account") do |element|
      @accounts << Account.from_xml(element)
    end
  end
  
  return @accounts
end
